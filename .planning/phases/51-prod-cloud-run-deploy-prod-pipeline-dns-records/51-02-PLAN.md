**Step 4 — Write `scripts/import-cloudflare-apex.sh` (one-time helper, `set -euo pipefail`):**

```bash
#!/usr/bin/env bash
# Bring the existing Cloudflare apex A record under Terraform management.
# MUST run after dns-prod.tf is committed but BEFORE `terraform apply`.
# Idempotent: re-running is safe (TF import complains "already managed" → harmless).
set -euo pipefail

: "${CLOUDFLARE_API_TOKEN:?set CLOUDFLARE_API_TOKEN env var}"
: "${CLOUDFLARE_ZONE_ID:?set CLOUDFLARE_ZONE_ID env var}"

# Resolve apex record ID via Cloudflare API
RECORD_ID=$(curl -sfS \
  -H "Authorization: Bearer $CLOUDFLARE_API_TOKEN" \
  "https://api.cloudflare.com/client/v4/zones/${CLOUDFLARE_ZONE_ID}/dns_records?type=A&name=nextlevelmock.com" \
  | jq -r '.result[0].id // empty')

[ -n "$RECORD_ID" ] || { echo "apex A record not found in zone $CLOUDFLARE_ZONE_ID" >&2; exit 1; }

echo "importing cloudflare_record.apex (record id: $RECORD_ID, zone id: $CLOUDFLARE_ZONE_ID)"

# Import — Cloudflare provider format is <zone-id>/<record-id>
cd "$(dirname "$0")/.."
terraform import \
  -var-file=prod.tfvars \
  cloudflare_record.apex "${CLOUDFLARE_ZONE_ID}/${RECORD_ID}" || {
    echo "import failed — if error is 'already imported', that's fine and harmless" >&2
    exit 0
  }

echo "imported OK. Run terraform plan to verify zero diff."
```

**`scripts/verify-dns-records.sh`** (bash, `set -euo pipefail`):

```bash
#!/usr/bin/env bash
# Phase 51 DNS-02 verification.
# Asserts the 4-record matrix in CONTEXT D-01 resolves correctly against Cloudflare's 1.1.1.1 resolver.
#
# CRITICAL NEGATIVE ASSERTION (T-51-01):
# apex must STILL resolve to v0.1 GCE IP — if it resolves to the prod Cloud Run LB IP,
# Phase 51 has accidentally performed the Phase 52 cutover. Script exits 2 in that case.
set -euo pipefail

cd "$(dirname "$0")/.."

V01_IP=$(grep -E '^v01_gce_ip' prod.tfvars | sed -E 's/.*"([^"]+)".*/\1/')
PROD_LB_IP=$(terraform output -raw prod_lb_ip)
STAGING_LB_IP=$(terraform output -raw staging_lb_ip 2>/dev/null || echo "")

[ -n "$V01_IP" ] || { echo "FAIL: v01_gce_ip not set in prod.tfvars" >&2; exit 1; }
[ -n "$PROD_LB_IP" ] || { echo "FAIL: prod_lb_ip terraform output missing (Plan 01 incomplete?)" >&2; exit 1; }

fail() { echo "FAIL: $*" >&2; exit 1; }
warn() { echo "WARN: $*" >&2; }
ok()   { echo "OK:   $*"; }

# Record 1: apex → v0.1 GCE IP (NEGATIVE ASSERTION — must NOT be prod LB IP)
APEX_RESOLVED=$(dig +short A nextlevelmock.com @1.1.1.1 | head -1)
if [ "$APEX_RESOLVED" = "$PROD_LB_IP" ]; then
  echo "FAIL(T-51-01 TRIGGERED): apex resolves to prod LB IP ($PROD_LB_IP) — cutover happened early!" >&2
  exit 2
fi
# Apex is Cloudflare-proxied so dig returns a CF anycast IP, not the origin IP directly.
# Instead, query Cloudflare API for the record's underlying value:
CF_APEX_VALUE=$(curl -sfS \
  -H "Authorization: Bearer $CLOUDFLARE_API_TOKEN" \
  "https://api.cloudflare.com/client/v4/zones/${CLOUDFLARE_ZONE_ID}/dns_records?type=A&name=nextlevelmock.com" \
  | jq -r '.result[0].content')
[ "$CF_APEX_VALUE" = "$V01_IP" ] || fail "apex Cloudflare record value = $CF_APEX_VALUE, expected $V01_IP (v0.1 GCE)"
ok "apex record value = $V01_IP (v0.1 GCE) — NOT flipped to prod"

# Record 2: www → prod Cloud Run LB IP (proxied=false so dig returns origin directly)
WWW_RESOLVED=$(dig +short A www.nextlevelmock.com @1.1.1.1 | head -1)
[ "$WWW_RESOLVED" = "$PROD_LB_IP" ] || fail "www resolves to $WWW_RESOLVED, expected $PROD_LB_IP"
ok "www.nextlevelmock.com → $PROD_LB_IP (prod Cloud Run LB)"

# Record 3: legacy → v0.1 GCE IP (proxied=false)
LEGACY_RESOLVED=$(dig +short A legacy.nextlevelmock.com @1.1.1.1 | head -1)
[ "$LEGACY_RESOLVED" = "$V01_IP" ] || fail "legacy resolves to $LEGACY_RESOLVED, expected $V01_IP"
ok "legacy.nextlevelmock.com → $V01_IP (v0.1 GCE — 30-day rollback record)"

# Record 4: staging → staging Cloud Run LB IP (Phase 47 record — unchanged)
STAGING_RESOLVED=$(dig +short A staging.nextlevelmock.com @1.1.1.1 | head -1)
if [ -n "$STAGING_LB_IP" ]; then
  [ "$STAGING_RESOLVED" = "$STAGING_LB_IP" ] || fail "staging resolves to $STAGING_RESOLVED, expected $STAGING_LB_IP"
  ok "staging.nextlevelmock.com → $STAGING_LB_IP (Phase 47)"
else
  [ -n "$STAGING_RESOLVED" ] || fail "staging record missing from DNS"
  warn "staging_lb_ip output missing — resolved value: $STAGING_RESOLVED (manual confirmation needed)"
fi

echo ""
echo "All 4 Phase 51 DNS records verified."
```
