Create `.github/workflows/deploy-staging.yml` with exactly this structure. Every line is deliberate — do NOT rearrange or "simplify".

```yaml
name: Deploy Staging

on:
  push:
    branches: [main]
  workflow_dispatch: {}

concurrency:
  group: deploy-staging
  cancel-in-progress: false

permissions:
  id-token: write    # Required for WIF OIDC token minting (T-48-04 scoped per-job)
  contents: read

jobs:
  deploy:
    name: Build + Migrate + Deploy + Smoke
    runs-on: ubuntu-latest
    timeout-minutes: 15
    env:
      PROJECT_ID: nlm-staging-493715
      REGION: us-central1
      SERVICE: nlm-staging
      IMAGE_REPO: us-central1-docker.pkg.dev/nlm-staging-493715/nlm-app/nlm-app
      RUNTIME_SA: nlm-cloudrun-sa@nlm-staging-493715.iam.gserviceaccount.com
      HEALTHCHECK_URL: https://staging.nextlevelmock.com/api/health

    steps:
      # ---- STEP 1: Checkout + WIF auth ------------------------------------------------
      - name: Checkout
        uses: actions/checkout@v4

      - name: Authenticate to GCP via WIF
        id: auth
        uses: google-github-actions/auth@v2
        with:
          workload_identity_provider: projects/${{ vars.STAGING_PROJECT_NUMBER }}/locations/global/workloadIdentityPools/github-actions/providers/github
          service_account: github-actions-deployer@nlm-staging-493715.iam.gserviceaccount.com

      - name: Set up gcloud
        uses: google-github-actions/setup-gcloud@v2
        with:
          project_id: ${{ env.PROJECT_ID }}

      # ---- STEP 2: Docker build + push to Artifact Registry --------------------------
      # D-21: deploy by digest (emitted by docker/build-push-action).
      # D-06: type=gha layer cache.
      - name: Configure Docker auth for Artifact Registry
        run: gcloud auth configure-docker us-central1-docker.pkg.dev --quiet

      - name: Set up Docker Buildx
        uses: docker/setup-buildx-action@v3

      - name: Build + push image
        id: build
        uses: docker/build-push-action@v5
        with:
          context: .
          push: true
          tags: |
            ${{ env.IMAGE_REPO }}:latest
            ${{ env.IMAGE_REPO }}:${{ github.sha }}
          cache-from: type=gha
          cache-to: type=gha,mode=max
          provenance: false   # avoid dual-digest manifest (simplifies @sha256 pin)

      - name: Verify digest is non-empty
        run: |
          [ -n "${{ steps.build.outputs.digest }}" ] \
            || (echo "::error::docker/build-push-action did not emit a digest"; exit 1)
          echo "Image digest: ${{ steps.build.outputs.digest }}"

      # ---- STEP 3: Fetch DIRECT_URL from Secret Manager (D-19) -----------------------
      - name: Fetch DIRECT_URL from Secret Manager
        id: secrets
        uses: google-github-actions/get-secretmanager-secrets@v2
        with:
          secrets: |-
            DIRECT_URL:${{ env.PROJECT_ID }}/DIRECT_URL

      # ---- STEP 4: Prisma migrate deploy (D-20: BEFORE gcloud run deploy) ------------
      - name: Set up Node 22
        uses: actions/setup-node@v4
        with:
          node-version: '22'
          cache: 'npm'

      - name: Install deps (minimal for prisma)
        run: npm ci --prefer-offline --no-audit --no-fund

      - name: Prisma migrate deploy against staging DIRECT_URL
        env:
          DATABASE_URL: ${{ steps.secrets.outputs.DIRECT_URL }}
        run: npx prisma migrate deploy

      # ---- STEP 5: Deploy Cloud Run by digest (D-21) ---------------------------------
      - name: Deploy to Cloud Run
        id: deploy
        run: |
          gcloud run deploy ${{ env.SERVICE }} \
            --project=${{ env.PROJECT_ID }} \
            --region=${{ env.REGION }} \
            --image=${{ env.IMAGE_REPO }}@${{ steps.build.outputs.digest }} \
            --service-account=${{ env.RUNTIME_SA }} \
            --quiet
          REVISION=$(gcloud run services describe ${{ env.SERVICE }} \
            --project=${{ env.PROJECT_ID }} \
            --region=${{ env.REGION }} \
            --format='value(status.latestCreatedRevisionName)')
          echo "revision=$REVISION" >> "$GITHUB_OUTPUT"

      # ---- STEP 6: Post-deploy smoke (D-02 step 4) -----------------------------------
      # Accepts 200 OR 503 per INFRA-07 Option C / Phase 47 D-24 (Prisma may be transiently down).
      - name: Smoke /api/health
        run: |
          for i in 1 2 3 4 5 6; do
            CODE=$(curl -s -o /dev/null -w '%{http_code}' -m 30 "${{ env.HEALTHCHECK_URL }}" || echo "000")
            echo "attempt $i → HTTP $CODE"
            if [ "$CODE" = "200" ] || [ "$CODE" = "503" ]; then
              echo "::notice::Smoke check passed (HTTP $CODE; 503 accepted per INFRA-07 Option C)"
              exit 0
            fi
            sleep 10
          done
          echo "::error::Smoke check failed after 6 retries"
          exit 1

      # ---- STEP 7: Job summary -------------------------------------------------------
      - name: Job summary
        if: always()
        run: |
          {
            echo "## Deploy Staging — ${{ job.status }}"
            echo ""
            echo "- Image: \`${{ env.IMAGE_REPO }}@${{ steps.build.outputs.digest }}\`"
            echo "- Revision: \`${{ steps.deploy.outputs.revision }}\`"
            echo "- Health URL: ${{ env.HEALTHCHECK_URL }}"
          } >> "$GITHUB_STEP_SUMMARY"
```

- D-06 Docker layer cache: `type=gha` with `mode=max`; npm cache via setup-node.

- Run migrate deploy with `DATABASE_URL` (pooler URL) — must use `DIRECT_URL` per Phase 46 D-12 + Prisma 7 advisory-lock requirement.

```yaml
name: Rollback Cloud Run (Manual)

on:
  workflow_dispatch:
    inputs:
      env:
        description: 'Target environment (staging | prod)'
        required: true
        type: choice
        options: [staging, prod]
        default: staging
      revision:
        description: 'Revision to pin to 100% traffic (e.g. nlm-staging-00042-abc). Run `gcloud run revisions list` to find the previous revision name.'
        required: true
        type: string

concurrency:
  group: rollback-${{ inputs.env }}
  cancel-in-progress: false

permissions:
  id-token: write
  contents: read

jobs:
  rollback:
    name: Rollback ${{ inputs.env }} to ${{ inputs.revision }}
    runs-on: ubuntu-latest
    timeout-minutes: 10
    steps:
      - name: Resolve target config
        id: cfg
        run: |
          if [ "${{ inputs.env }}" = "prod" ]; then
            echo "project_id=nlm-prod" >> "$GITHUB_OUTPUT"
            echo "service=nlm-prod" >> "$GITHUB_OUTPUT"
            echo "project_number=${{ vars.PROD_PROJECT_NUMBER }}" >> "$GITHUB_OUTPUT"
            echo "sa=github-actions-deployer@nlm-prod.iam.gserviceaccount.com" >> "$GITHUB_OUTPUT"
          else
            echo "project_id=nlm-staging-493715" >> "$GITHUB_OUTPUT"
            echo "service=nlm-staging" >> "$GITHUB_OUTPUT"
            echo "project_number=${{ vars.STAGING_PROJECT_NUMBER }}" >> "$GITHUB_OUTPUT"
            echo "sa=github-actions-deployer@nlm-staging-493715.iam.gserviceaccount.com" >> "$GITHUB_OUTPUT"
          fi

      - name: Authenticate to GCP via WIF
        uses: google-github-actions/auth@v2
        with:
          workload_identity_provider: projects/${{ steps.cfg.outputs.project_number }}/locations/global/workloadIdentityPools/github-actions/providers/github
          service_account: ${{ steps.cfg.outputs.sa }}

      - name: Set up gcloud
        uses: google-github-actions/setup-gcloud@v2
        with:
          project_id: ${{ steps.cfg.outputs.project_id }}

      - name: Verify revision exists
        run: |
          gcloud run revisions describe "${{ inputs.revision }}" \
            --project=${{ steps.cfg.outputs.project_id }} \
            --region=us-central1 \
            --service=${{ steps.cfg.outputs.service }} > /dev/null

      - name: Pin traffic to revision
        run: |
          gcloud run services update-traffic "${{ steps.cfg.outputs.service }}" \
            --project=${{ steps.cfg.outputs.project_id }} \
            --region=us-central1 \
            --to-revisions="${{ inputs.revision }}=100" \
            --quiet

      - name: Verify pinned revision
        run: |
          PINNED=$(gcloud run services describe "${{ steps.cfg.outputs.service }}" \
            --project=${{ steps.cfg.outputs.project_id }} \
            --region=us-central1 \
            --format='value(status.traffic[0].revisionName)')
          echo "Currently serving: $PINNED"
          [ "$PINNED" = "${{ inputs.revision }}" ] \
            || (echo "::error::Pinned revision mismatch"; exit 1)

      - name: Job summary
        if: always()
        run: |
          {
            echo "## Rollback — ${{ job.status }}"
            echo ""
            echo "- Environment: \`${{ inputs.env }}\`"
            echo "- Service: \`${{ steps.cfg.outputs.service }}\`"
            echo "- Pinned revision: \`${{ inputs.revision }}\`"
          } >> "$GITHUB_STEP_SUMMARY"
```

# GitHub Actions Repo Variables — NLM v1.5

**Purpose:** The deploy / rollback workflows reference GCP project numbers via `${{ vars.STAGING_PROJECT_NUMBER }}` and `${{ vars.PROD_PROJECT_NUMBER }}`. These must be set as **repo variables** (not secrets) before any workflow runs will succeed.

## Why Variables, Not Secrets?

GCP project numbers are **not sensitive**. They are:
- Visible to anyone with `roles/viewer` on the project.
- Required in error messages and WIF provider paths that must be logged for debugging.
- Static for the life of the project.

Using `${{ secrets.X }}` would needlessly mask them in logs, making failed runs harder to debug.

## Required Variables

| Variable | Value | Source |
|----------|-------|--------|
| `STAGING_PROJECT_NUMBER` | 12-digit project number for `nlm-staging-493715` | `gcloud projects describe nlm-staging-493715 --format='value(projectNumber)'` |
| `PROD_PROJECT_NUMBER` | 12-digit project number for `nlm-prod` | `gcloud projects describe nlm-prod --format='value(projectNumber)'` |

## How to Set (one-time)

### Option A — via `gh` CLI

```bash
STAGING_NUM=$(gcloud projects describe nlm-staging-493715 --format='value(projectNumber)')
PROD_NUM=$(gcloud projects describe nlm-prod --format='value(projectNumber)')

gh variable set STAGING_PROJECT_NUMBER --body "$STAGING_NUM" --repo JesterCharles/mock-interview-assist
gh variable set PROD_PROJECT_NUMBER --body "$PROD_NUM" --repo JesterCharles/mock-interview-assist
```

### Option B — via GitHub UI

1. Navigate to: https://github.com/JesterCharles/mock-interview-assist/settings/variables/actions
2. Click **New repository variable**
3. Name: `STAGING_PROJECT_NUMBER`, Value: (output of `gcloud projects describe nlm-staging-493715 --format='value(projectNumber)'`)
4. Repeat for `PROD_PROJECT_NUMBER`

## Verification

```bash
gh variable list --repo JesterCharles/mock-interview-assist
# Expected output includes STAGING_PROJECT_NUMBER and PROD_PROJECT_NUMBER
```

A failed-fast `deploy-staging.yml` run before these are set surfaces the error:
```
Error: failed to get ID token: could not find workload_identity_provider
```

## Rollback

```bash

gh variable delete PROD_PROJECT_NUMBER --repo JesterCharles/mock-interview-assist
```
