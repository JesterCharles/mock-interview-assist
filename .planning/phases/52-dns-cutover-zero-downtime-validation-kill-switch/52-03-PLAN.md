    Script contract (committed at `scripts/kill-switch.sh`, chmod +x):

    ```bash
    #!/usr/bin/env bash
    # scripts/kill-switch.sh — Fast DNS cutover / rollback for nextlevelmock.com apex.
    # Usage: kill-switch.sh {status|revert|restore}
    #   status   — Print current apex A record value.
    #   revert   — Flip apex A to V01_GCE_IP (emergency rollback to v0.1).
    #   restore  — Flip apex A to PROD_CLOUDRUN_LB_IP (forward to prod Cloud Run).
    # Requires env: CLOUDFLARE_API_TOKEN, CLOUDFLARE_ZONE_ID, V01_GCE_IP, PROD_CLOUDRUN_LB_IP.
    # Does NOT touch terraform state — operator MUST run `terraform refresh`
    # afterwards to reconcile iac/cloudrun/prod.tfvars with Cloudflare reality.
    #
    # Per Phase 52 D-11: chosen over `terraform apply -target=...` because this
    # bypasses the TF state round-trip (~30s+) in favor of a direct API PATCH (~2s).

    set -euo pipefail

    : "${CLOUDFLARE_API_TOKEN:?missing}"
    : "${CLOUDFLARE_ZONE_ID:?missing}"
    : "${V01_GCE_IP:?missing}"
    : "${PROD_CLOUDRUN_LB_IP:?missing}"

    CF_API="https://api.cloudflare.com/client/v4"
    AUTH_HEADER="Authorization: Bearer $CLOUDFLARE_API_TOKEN"

    # Look up apex record ID once (cached per invocation).
    get_apex_record_id() {
      curl -sf -H "$AUTH_HEADER" \
        "$CF_API/zones/$CLOUDFLARE_ZONE_ID/dns_records?type=A&name=nextlevelmock.com" \
        | jq -r '.result[0].id'
    }

    cmd_status() {
      curl -sf -H "$AUTH_HEADER" \
        "$CF_API/zones/$CLOUDFLARE_ZONE_ID/dns_records?type=A&name=nextlevelmock.com" \
        | jq -r '.result[0] | {name, content, proxied, modified_on}'
    }

    set_apex_value() {
      local new_ip="$1"
      local record_id
      record_id=$(get_apex_record_id)
      curl -sf -X PATCH -H "$AUTH_HEADER" -H 'Content-Type: application/json' \
        "$CF_API/zones/$CLOUDFLARE_ZONE_ID/dns_records/$record_id" \
        -d "$(jq -cn --arg ip "$new_ip" '{content:$ip, type:"A", name:"nextlevelmock.com", proxied:true}')" \
        | jq -r '.result | {name, content, proxied, modified_on}'
    }

    cmd_revert()  { set_apex_value "$V01_GCE_IP"; echo "reverted to v0.1"; }
    cmd_restore() { set_apex_value "$PROD_CLOUDRUN_LB_IP"; echo "restored to prod Cloud Run"; }

    case "${1:-}" in
      status)  cmd_status  ;;
      revert)  cmd_revert  ;;
      restore) cmd_restore ;;
      *) echo "usage: $0 {status|revert|restore}"; exit 2 ;;
    esac
    ```
