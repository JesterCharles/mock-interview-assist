   ```ts
   const res = await fetch('/api/cohorts')
   if (res.ok) {
     const data: Array<{ id: string; name: string; startDate?: string }> = await res.json()
     // Sort by startDate desc per D-03
     data.sort((a, b) => (b.startDate ?? '').localeCompare(a.startDate ?? ''))
     setCohorts(data.map(c => ({ id: c.id, name: c.name })))
   }
   ```
