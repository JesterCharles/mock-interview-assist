    // Trainer-only prefixes (existing, unchanged behavior for trainer holders)
    const TRAINER_PATHS = ['/dashboard', '/interview', '/review', '/trainer'];
    // Associate-accessible prefixes (also accepts trainer — trainers can view any associate)
    const ASSOCIATE_PATHS = ['/associate'];
    // Public carve-out within associate namespace
    const PUBLIC_ASSOCIATE_PATHS = ['/associate/login'];
