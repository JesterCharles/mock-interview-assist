```typescript
import crypto from 'crypto';

const resend = new Resend(process.env.RESEND_API_KEY || 're_dummy_key');
const SITE = process.env.NEXT_PUBLIC_SITE_URL ?? 'http://localhost:3000';
const THROTTLE_MS = 5 * 60 * 1000; // 5 minutes

export type InviteResult = {
  status: 'invited' | 'reassigned' | 'skipped' | 'failed';
  error?: string;
};

function generateSlug(email: string): string {
  const local = email.split('@')[0].replace(/[^a-z0-9]/gi, '').toLowerCase().slice(0, 20);
  const suffix = crypto.randomBytes(2).toString('hex'); // 4 hex chars
  return `${local || 'user'}-${suffix}`;
}

export async function inviteAssociate(
  email: string,
  cohortId: number,
  trainerIdentifier: string,
): Promise<InviteResult> { ... }
```

Function logic (sequential, strict ordering per Research Pitfall 5):

1. **Upsert Associate by email**: `prisma.associate.upsert({ where: { email }, update: { cohortId }, create: { email, slug: generateSlug(email), cohortId } })`. Catch P2002 on slug collision -- retry once with new suffix.

2. **Detect action type**: If the upsert matched an existing record:
   - If `associate.cohortId` was already `cohortId` before update: return `{ status: 'skipped', error: 'Already in target cohort' }`
   - Need to check: query `associate` before upsert OR use a separate findUnique + conditional update. Better approach: `findUnique({ where: { email } })` first, then decide:
     - Not found -> create -> status will be 'invited'
     - Found, same cohortId -> return 'skipped'
     - Found, different cohortId -> update cohortId -> status will be 'reassigned'
     - Found, lastInvitedAt < 5 min -> return 'skipped' with throttle message

3. **Check 5-min throttle**: If existing associate has `lastInvitedAt` and `Date.now() - lastInvitedAt.getTime() < THROTTLE_MS`, return `{ status: 'skipped', error: 'Recently invited -- throttled' }`.

4. **generateLink**: `supabaseAdmin.auth.admin.generateLink({ type: 'magiclink', email, options: { redirectTo: '${SITE}/auth/callback' } })`. If error -> return `{ status: 'failed', error: linkError.message }`.

5. **Resend send**: `resend.emails.send({ from: 'Next Level Mock <noreply@nextlevelmock.com>', to: email, subject: "You're invited to Next Level Mock", html: getMagicLinkEmailHtml(linkData.properties.action_link) })`. If error -> return `{ status: 'failed', error: String(emailErr) }`.

6. **Update lastInvitedAt**: `prisma.associate.update({ where: { email }, data: { lastInvitedAt: new Date() } })`.

7. **Record auth event**: `await recordAuthEvent({ type: 'trainer-invite', email, ip: trainerIdentifier })`.

8. Return `{ status: isNew ? 'invited' : 'reassigned' }`.

Also export `generateSlug` for testing.

Create `src/app/api/trainer/invites/bulk/route.ts`:

```typescript
import { NextRequest, NextResponse } from 'next/server';
import { z } from 'zod';
import { getCallerIdentity } from '@/lib/identity';
import { prisma } from '@/lib/prisma';
import { inviteAssociate } from '@/lib/inviteHelper';

const DAILY_INVITE_LIMIT = 20;

const BulkInviteSchema = z.object({
  emails: z.array(z.string().email()).min(1).max(50),
  cohortId: z.number().int().positive(),
});
```

Route handler logic:

1. **Auth gate**: `getCallerIdentity()` -- reject if `caller.kind !== 'trainer' && caller.kind !== 'admin'` -> 401.

2. **Validate body**: `BulkInviteSchema.safeParse(body)` -- if fails -> 400 with `error.flatten()`.

3. **Verify cohort exists**: `prisma.cohort.findUnique({ where: { id: cohortId } })` -- if not found -> 404 `"Cohort not found"`.

4. **Pre-flight daily limit** (per D-13):
   ```typescript
   const since = new Date(Date.now() - 24 * 60 * 60 * 1000);
   const todayCount = await prisma.authEvent.count({
     where: {
       type: 'trainer-invite',
       ip: caller.email ?? 'trainer',
       createdAt: { gte: since },
     },
   });
   const remaining = DAILY_INVITE_LIMIT - todayCount;
   if (remaining < emails.length) {
     return NextResponse.json(
       { error: `Would exceed daily limit (${remaining} remaining of ${DAILY_INVITE_LIMIT})` },
       { status: 429 }
     );
   }
   ```
   Note: filter by `ip` field = `caller.email ?? 'trainer'` to scope per-trainer (matches single-invite pattern where `ip` stores trainer identifier).

5. **Sequential processing** (per D-14):
   ```typescript
   const results: { email: string; status: string; error?: string }[] = [];
   for (const email of emails) {
     try {
       const result = await inviteAssociate(email, cohortId, caller.email ?? 'trainer');
       results.push({ email, ...result });
     } catch (err) {
       results.push({ email, status: 'failed', error: String(err) });
     }
   }
   ```

6. **Return** (per D-15): `NextResponse.json({ results })` with status 200.
