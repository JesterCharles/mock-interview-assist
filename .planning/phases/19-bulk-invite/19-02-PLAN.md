```typescript
import type { ParsedEmail } from '@/lib/emailParser';

export type PreviewAction = 'new' | 'reassign' | 'skip-same-cohort' | 'skip-recently-invited' | 'skip-invalid';

export interface PreviewRow {
  email: string;
  action: PreviewAction;
  actionLabel: string;  // display text per D-09
  notes: string;
  checkable: boolean;   // only 'new' and 'reassign' are checkable
  checked: boolean;     // default true for checkable rows
}

export interface RosterAssociate {
  email: string | null;
  cohortId: number | null;
  lastInvitedAt: string | null; // ISO string
  slug: string;
}

export interface CohortOption {
  id: number;
  name: string;
}
```

Function `classifyEmails(chips: ParsedEmail[], roster: RosterAssociate[], targetCohortId: number, cohorts: CohortOption[]): PreviewRow[]`:

For each chip:
- If `chip.state === 'invalid'` -> `{ action: 'skip-invalid', actionLabel: 'Skip -- invalid email', notes: '', checkable: false, checked: false }`
- If `chip.state === 'duplicate'` -> skip entirely (don't include in preview)
- Find matching roster entry by `chip.normalized === rosterEntry.email?.toLowerCase()`
- If not found -> `{ action: 'new', actionLabel: 'New', notes: '', checkable: true, checked: true }`
- If found and `rosterEntry.cohortId === targetCohortId` -> `{ action: 'skip-same-cohort', actionLabel: 'Skip -- same cohort', notes: '', checkable: false, checked: false }`
- If found and `rosterEntry.lastInvitedAt` within 5 min of now -> `{ action: 'skip-recently-invited', actionLabel: 'Skip -- recently invited', notes: 'Invited N min ago', checkable: false, checked: false }`
- If found and different cohortId -> `{ action: 'reassign', actionLabel: 'Reassign cohort', notes: 'Currently in {cohortName}', checkable: true, checked: true }`
  - Look up current cohort name from `cohorts` array by `rosterEntry.cohortId`

Write tests FIRST in `src/lib/bulkInvitePreview.test.ts`. Cover all 6 classification branches.
