    **Step 2 — Create `scripts/judge0-spike.ts`**:

    ```typescript
    #!/usr/bin/env tsx
    // Spike harness: JUDGE-06 gate (Plan 38-03).
    // Submits 10 fixtures in parallel, polls each to completion,
    // samples `docker stats` every 1 sec, emits report with peaks + p50/p95.
    //
    // Usage:
    //   JUDGE0_URL=http://localhost:2358 JUDGE0_AUTH_TOKEN=xxx tsx scripts/judge0-spike.ts
    //   Optional: RUNS=3 (default), OUTPUT=scripts/judge0-spike-output.json

    import { submit, getSubmission, type Judge0Language } from '../src/lib/judge0Client';
    import { readdir, readFile, writeFile } from 'fs/promises';
    import { spawn } from 'child_process';
    import path from 'path';

    interface Fixture {
      name: string;
      language: Judge0Language;
      sourceCode: string;
      stdin: string;
      expectedStdout: string;
      maxWallTimeSec: number;
    }

    const FIXTURE_DIR = path.join(__dirname, 'judge0-spike-fixtures');
    const RUNS = Number(process.env.RUNS ?? '3');

    async function loadFixtures(): Promise<Fixture[]> {
      const files = (await readdir(FIXTURE_DIR)).filter(f => f.endsWith('.json'));
      return Promise.all(files.map(async f =>
        JSON.parse(await readFile(path.join(FIXTURE_DIR, f), 'utf-8'))
      ));
    }

    async function pollUntilDone(token: string, maxSec = 60) {
      const start = Date.now();
      while ((Date.now() - start) / 1000 < maxSec) {
        const res = await getSubmission(token);
        if (res.status.id >= 3) return res;
        await new Promise(r => setTimeout(r, 500));
      }
      throw new Error(`Poll timeout for ${token}`);
    }

    function normalizeOutput(s: string | null): string {
      return (s ?? '').replace(/\r\n/g, '\n').trimEnd() + '\n';
    }

    function startDockerStatsSampler() {
      const samples: SampleRow[] = [];
      const child = spawn('docker', ['stats', '--no-trunc', '--format',
        '{{.Name}}|{{.CPUPerc}}|{{.MemUsage}}|{{.MemPerc}}'], { stdio: ['ignore', 'pipe', 'ignore'] });
      const start = Date.now();
      child.stdout.on('data', buf => {
        for (const line of buf.toString().split('\n').filter(Boolean)) {
          const [name, cpuPerc, memUsage] = line.split('|');
          if (!name?.startsWith('judge0')) continue;
          samples.push({
            timestampSec: (Date.now() - start) / 1000,
            container: name.trim(),
            cpuPct: parseFloat(cpuPerc) || 0,
            memMiB: parseMemUsage(memUsage),
          });
        }
      });
      return {
        stop: async () => { child.kill('SIGTERM'); await new Promise(r => setTimeout(r, 200)); return samples; },
      };
    }

    function parseMemUsage(s: string): number {
      const [used] = s.trim().split('/');
      const n = parseFloat(used);
      if (used.includes('GiB')) return n * 1024;
      return n;
    }

    function percentile(xs: number[], p: number): number {
      if (xs.length === 0) return 0;
      const sorted = [...xs].sort((a, b) => a - b);
      return sorted[Math.min(sorted.length - 1, Math.floor(sorted.length * p))];
    }

    async function runOnce(runId: number, fixtures: Fixture[]) {
      const sampler = startDockerStatsSampler();
      const wallStart = Date.now();
      const submitStarts: Record<string, number> = {};

      const submits = await Promise.all(fixtures.map(async fx => {
        submitStarts[fx.name] = Date.now();
        const { token } = await submit({
          sourceCode: fx.sourceCode,
          language: fx.language,
          stdin: fx.stdin,
          expectedStdout: fx.expectedStdout,
        });
        return { fx, token };
      }));

      const results = await Promise.all(submits.map(async ({ fx, token }) => {
        const res = await pollUntilDone(token, fx.maxWallTimeSec * 3);
        const completedAt = Date.now();
        const correct = normalizeOutput(res.stdout) === normalizeOutput(fx.expectedStdout) && res.status.id === 3;
        return {
          fixture: fx.name,
          language: fx.language,
          token,
          submittedAt: new Date(submitStarts[fx.name]).toISOString(),
          completedAt: new Date(completedAt).toISOString(),
          latencySec: (completedAt - submitStarts[fx.name]) / 1000,
          verdict: res.status.description,
          expectedVerdict: 'pass' as const,
          correct,
        };
      }));

      const wallClockSec = (Date.now() - wallStart) / 1000;
      const samples = await sampler.stop();
      const peaks: Record<string, { cpuPct: number; memMiB: number }> = {};
      for (const s of samples) {
        const p = peaks[s.container] ?? { cpuPct: 0, memMiB: 0 };
        peaks[s.container] = { cpuPct: Math.max(p.cpuPct, s.cpuPct), memMiB: Math.max(p.memMiB, s.memMiB) };
      }
      const byLang: Record<string, number[]> = {};
      for (const r of results) (byLang[r.language] ??= []).push(r.latencySec);
      const p50: Record<string, number> = {};
      const p95: Record<string, number> = {};
      for (const [lang, xs] of Object.entries(byLang)) { p50[lang] = percentile(xs, 0.5); p95[lang] = percentile(xs, 0.95); }
      return { runId, wallClockSec, submissions: results, dockerStats: samples, peaks, p50Latency: p50, p95Latency: p95 };
    }

    async function main() {
      const fixtures = await loadFixtures();
      if (fixtures.length !== 10) throw new Error(`Expected 10 fixtures, got ${fixtures.length}`);
      const runs = [];
      for (let i = 1; i <= RUNS; i++) {
        console.log(`[spike] Run ${i}/${RUNS} starting...`);
        runs.push(await runOnce(i, fixtures));
        if (i < RUNS) await new Promise(r => setTimeout(r, 10_000));
      }
      const output = process.env.OUTPUT ?? 'scripts/judge0-spike-output.json';
      await writeFile(output, JSON.stringify({ runs, producedAt: new Date().toISOString() }, null, 2));
      for (const r of runs) {
        const allCorrect = r.submissions.every(s => s.correct);
        console.log(`  Run ${r.runId}: wall=${r.wallClockSec.toFixed(2)}s, allCorrect=${allCorrect}, peaks=${JSON.stringify(r.peaks)}`);
      }
    }

    type SampleRow = { timestampSec: number; container: string; cpuPct: number; memMiB: number };

    main().catch(err => { console.error(err); process.exit(1); });
    ```
