    ```typescript
    // src/lib/judge0Client.ts
    import { Judge0ConfigError, Judge0UnavailableError, UnsupportedLanguageError } from './judge0Errors';

    export type Judge0Language = 'python' | 'javascript' | 'typescript' | 'java' | 'sql' | 'csharp';

    export const JUDGE0_LANGUAGE_MAP: Record&lt;Judge0Language, number&gt; = {
      python: 71,
      javascript: 93,
      typescript: 94,
      java: 62,
      sql: 82,
      csharp: 51,
    };

    export interface SubmitOptions { /* ... */ }
    export interface Judge0Submission { /* ... */ }
    export interface Judge0SystemInfo { /* ... */ }

    function getEnv(): { url: string; token: string } {
      const url = process.env.JUDGE0_URL;
      const token = process.env.JUDGE0_AUTH_TOKEN;
      if (!url) throw new Judge0ConfigError('JUDGE0_URL');
      if (!token) throw new Judge0ConfigError('JUDGE0_AUTH_TOKEN');
      return { url: url.replace(/\/$/, ''), token };
    }

    async function withRetry&lt;T&gt;(fn: () =&gt; Promise&lt;T&gt;, backoffMs = 1000): Promise&lt;T&gt; {
      try { return await fn(); }
      catch (err) {
        if (!isRetryable(err)) throw err;
        await new Promise(r =&gt; setTimeout(r, backoffMs));
        return fn();
      }
    }

    function isRetryable(err: unknown): boolean {
      // AbortError, TypeError (fetch network error), or Judge0UnavailableError with 5xx cause
      // 4xx: NOT retryable
      // ...
    }

    export async function submit(opts: SubmitOptions): Promise&lt;{ token: string }&gt; {
      if (!(opts.language in JUDGE0_LANGUAGE_MAP)) {
        throw new UnsupportedLanguageError(opts.language);
      }
      const { url, token } = getEnv();
      return withRetry(async () =&gt; {
        const res = await fetch(`${url}/submissions?base64_encoded=false&amp;wait=false`, {
          method: 'POST',
          headers: { 'Content-Type': 'application/json', 'X-Auth-Token': token },
          body: JSON.stringify({
            source_code: opts.sourceCode,
            language_id: JUDGE0_LANGUAGE_MAP[opts.language],
            stdin: opts.stdin ?? '',
            expected_output: opts.expectedStdout ?? null,
            cpu_time_limit: opts.cpuTimeLimit ?? 10,
            memory_limit: opts.memoryLimit ?? 256000,
          }),
          signal: AbortSignal.timeout(30_000),
        });
        if (res.status &gt;= 400 &amp;&amp; res.status &lt; 500) {
          throw new Judge0UnavailableError(`HTTP ${res.status}: ${await res.text()}`);
        }
        if (!res.ok) throw new Judge0UnavailableError(`HTTP ${res.status}`);
        return res.json();
      });
    }

    export async function getSubmission(token: string): Promise&lt;Judge0Submission&gt; {
      const { url, token: authToken } = getEnv();
      return withRetry(async () =&gt; {
        const res = await fetch(`${url}/submissions/${encodeURIComponent(token)}?base64_encoded=false`, {
          headers: { 'X-Auth-Token': authToken },
          signal: AbortSignal.timeout(30_000),
        });
        if (!res.ok) throw new Judge0UnavailableError(`HTTP ${res.status}`);
        return res.json();
      });
    }

    export async function systemInfo(timeoutMs = 2000): Promise&lt;Judge0SystemInfo&gt; {
      const { url, token } = getEnv();
      // NO RETRY on systemInfo — health probe semantics.
      try {
        const res = await fetch(`${url}/system_info`, {
          headers: { 'X-Auth-Token': token },
          signal: AbortSignal.timeout(timeoutMs),
        });
        if (!res.ok) throw new Judge0UnavailableError(`HTTP ${res.status}`);
        return res.json();
      } catch (err) {
        throw new Judge0UnavailableError(err);
      }
    }
    ```
