    ## Auth gating (D-13)
    - anonymous → 401 {error:'unauthorized'}
    - associate → 403 {error:'forbidden'}
    - trainer → 200
    - admin → 200 (admin counts as trainer-or-higher)

    ## Body parsing
    - empty body (no payload) → full manifest sync (listChallenges called with no args)
    - `{}` → full manifest sync
    - `{slugs: []}` → full manifest sync (empty array treated as omitted)
    - `{slugs: ['two-sum']}` → only 'two-sum' synced; listChallenges NOT called
    - `{slugs: 'not-an-array'}` → 400 {error:'invalid_body'}
    - `{slugs: [42]}` → 400 (must be strings)
    - `{slugs: ['Not A Valid Slug']}` → 400 (slug regex)
    - malformed JSON body → 400

    ## Batch error isolation (core D-15 step 4 + CODING-BANK-05)
    - When slugs=['ok1', 'bad', 'ok2'] and syncChallengeToDb throws ChallengeValidationError for 'bad':
      - response is 200 (partial success, not 5xx)
      - body is `{synced: 2, skipped: 0, errors: [{slug: 'bad', reason: '...'}]}`
      - ok1 and ok2 completed successfully (syncChallengeToDb called for all 3)
    - When syncChallengeToDb throws a NON-ChallengeValidationError (e.g., network error):
      - error is still captured in errors[] with a sanitized `reason` (no stack trace string)
      - batch continues

    ## Invalidate before full sync
    - On full manifest sync (no slugs arg), `invalidateCache()` is called BEFORE listChallenges(). Verified by mock ordering.
    - On targeted slug sync, `invalidateCache('public:${slug}:*')` AND `invalidateCache('private:${slug}:*')` are called for each slug before syncChallengeToDb.

    ## Response ordering / determinism
    - errors[] entries appear in input-order (not set-order); synced count is a plain number
    - headers: `Content-Type: application/json`

    ## Edge cases
    - listChallenges itself throws ChallengeValidationError (e.g., duplicate slug in manifest) → response is 200 with synced=0 and errors=[{slug:'manifest', reason:'duplicate slug "x"'}] — do NOT return 500 for manifest-level validation
    - listChallenges throws a non-ChallengeValidationError (e.g., GitHub 503) → 502 {error:'upstream_unavailable', message:'...'}

    ## Step 2 — Implement `src/app/api/coding/bank/refresh/route.ts` (GREEN)

    ```typescript
    import 'server-only';
    import { NextRequest, NextResponse } from 'next/server';
    import { z } from 'zod';
    import { getCallerIdentity } from '@/lib/identity';
    import {
      listChallenges,
      syncChallengeToDb,
      invalidateCache,
    } from '@/lib/coding-challenge-service';
    import { ChallengeValidationError } from '@/lib/coding-bank-schemas';

    // Slug format must match curriculum/bank convention.
    const SLUG_REGEX = /^[a-z0-9][a-z0-9-]*$/;

    const BodySchema = z
      .object({
        slugs: z
          .array(z.string().regex(SLUG_REGEX).max(50))
          .optional(),
      })
      .strict();

    type ErrorEntry = { slug: string; reason: string };

    function sanitizeReason(err: unknown): string {
      if (err instanceof ChallengeValidationError) {
        return `${err.path}: ${err.reason}`;
      }
      if (err instanceof Error) {
        // First line only — keep stack traces out of API responses.
        return err.message.split('\n')[0].slice(0, 500);
      }
      return 'Unknown error';
    }

    export async function POST(req: NextRequest): Promise<NextResponse> {
      // 1. Auth gate (D-13) — trainer or admin only.
      const identity = await getCallerIdentity();
      if (identity.kind === 'anonymous') {
        return NextResponse.json(
          { error: 'unauthorized', message: 'Sign-in required' },
          { status: 401 },
        );
      }
      if (identity.kind !== 'trainer' && identity.kind !== 'admin') {
        return NextResponse.json(
          { error: 'forbidden', message: 'Trainer role required' },
          { status: 403 },
        );
      }

      // 2. Parse body (optional; empty body = full sync).
      let parsed: z.infer<typeof BodySchema> = {};
      if (req.headers.get('content-length') && req.headers.get('content-length') !== '0') {
        let raw: unknown;
        try {
          raw = await req.json();
        } catch {
          return NextResponse.json(
            { error: 'invalid_body', message: 'Body must be valid JSON' },
            { status: 400 },
          );
        }
        const result = BodySchema.safeParse(raw);
        if (!result.success) {
          return NextResponse.json(
            { error: 'invalid_body', message: result.error.issues[0]?.message ?? 'Invalid body' },
            { status: 400 },
          );
        }
        parsed = result.data;
      }

      const targetSlugs = parsed.slugs && parsed.slugs.length > 0 ? parsed.slugs : null;
      const errors: ErrorEntry[] = [];
      let synced = 0;
      const skipped = 0; // v1.5: unchanged-since-etag detection

      if (targetSlugs === null) {
        // Full-manifest path — flush cache first so the walk sees fresh GitHub state.
        invalidateCache();
        let manifest: Awaited<ReturnType<typeof listChallenges>>;
        try {
          manifest = await listChallenges();
        } catch (err) {
          if (err instanceof ChallengeValidationError) {
            // Manifest-level validation failure (e.g., duplicate slug) — surface as error entry, not 5xx.
            return NextResponse.json({
              synced: 0,
              skipped: 0,
              errors: [{ slug: err.slug ?? 'manifest', reason: sanitizeReason(err) }],
            });
          }
          // Upstream failure (GitHub down, etc.).
          return NextResponse.json(
            { error: 'upstream_unavailable', message: sanitizeReason(err) },
            { status: 502 },
          );
        }
        for (const entry of manifest) {
          try {
            await syncChallengeToDb(entry.slug);
            synced++;
          } catch (err) {
            errors.push({ slug: entry.slug, reason: sanitizeReason(err) });
          }
        }
      } else {
        // Targeted path — per-slug cache flush preserves input ordering.
        for (const slug of targetSlugs) {
          invalidateCache(`public:${slug}:*`);
          invalidateCache(`private:${slug}:*`);
          try {
            await syncChallengeToDb(slug);
            synced++;
          } catch (err) {
            errors.push({ slug, reason: sanitizeReason(err) });
          }
        }
      }

      return NextResponse.json({ synced, skipped, errors });
    }
    ```
