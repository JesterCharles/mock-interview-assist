   ```typescript
   import { z } from 'zod';

   export const slugSchema = z.string()
     .trim()
     .toLowerCase()
     .regex(/^[a-z0-9]+(?:-[a-z0-9]+)*$/, 'Must be lowercase letters, numbers, and hyphens only')
     .min(2, 'Slug must be at least 2 characters')
     .max(64, 'Slug must be 64 characters or fewer');

   export function validateSlug(raw: string): { success: true; slug: string } | { success: false; error: string } {
     const result = slugSchema.safeParse(raw);
     if (result.success) {
       return { success: true, slug: result.data };
     }
     return { success: false, error: result.error.issues[0]?.message ?? 'Invalid slug' };
   }
   ```
   NOTE: Check zod v4 API — `.transform()` may replace `.toLowerCase()`. Use whichever is correct for zod 4.3.6. The key constraint: output is always lowercased and trimmed.

## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| Client -> API | Slug value from dashboard wizard input crosses to /api/history POST |
| API -> Supabase | Associate upsert and session write cross to database |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-03-01 | Tampering | Slug input | mitigate | Server-side Zod validation in /api/history POST via validateSlug() before any DB operation. Client validation is UX only. |
| T-03-02 | Information Disclosure | /associate/[slug] enumeration | mitigate | Route requires isAuthenticatedSession() — unauthenticated requests get redirect to /login. Only trainers can view profiles. (Implemented in Plan 02.) |
| T-03-03 | Denial of Service | Overlong slug | mitigate | Zod slugSchema enforces .max(64) — rejects before DB query. |
| T-03-04 | Tampering | SQL injection via slug | mitigate | Prisma parameterized queries — slug never interpolated into raw SQL. Zod regex further constrains to `[a-z0-9-]` only. |
| T-03-05 | Repudiation | Anonymous session linked to wrong associate | accept | Low risk: trainer enters slug manually. If wrong slug entered, session links to wrong associate. No automated mitigation needed — trainer corrects on next session. |
</threat_model>
